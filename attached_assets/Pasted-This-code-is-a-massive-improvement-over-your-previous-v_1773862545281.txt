C. Full Proxy "Pipe" Improvements
Ensure your proxy handles Range headers properly. If the browser asks for bytes=0-100, your proxy must request exactly that from YouTube.

app.get('/api/proxy/:videoId', async (req, res) => {
  // ... select format logic ...
  
  const headers = {
    'User-Agent': YT_UA,
    'Referer': 'https://www.youtube.com'
  };
  
  // Forward the browser's Range header to YouTube
  if (req.headers.range) headers['Range'] = req.headers.range;

  const upstream = await fetch(format.url, { headers, signal: controller.signal });

  // Forward EXACT status (e.g., 206 Partial Content)
  res.status(upstream.status);
  
  // Pass through all critical streaming headers
  const passthrough = ['content-type', 'content-length', 'content-range', 'accept-ranges'];
  passthrough.forEach(h => {
    const v = upstream.headers.get(h);
    if (v) res.setHeader(h, v);
  });

  const body = Readable.fromWeb(upstream.body);
  body.pipe(res);
});
